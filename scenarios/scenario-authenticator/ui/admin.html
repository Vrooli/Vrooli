Quick Integration Guide

1. Discover Service Endpoints

Use the lifecycle utilities to fetch the dynamic ports allocated to the authenticator. Export the URLs so your services do not need to hard-code ports.

# From the Vrooli workspace root
AUTH_API_PORT=$(vrooli scenario port scenario-authenticator API_PORT)
AUTH_UI_PORT=$(vrooli scenario port scenario-authenticator UI_PORT)

export SCENARIO_AUTHENTICATOR_API_URL="http://localhost:${AUTH_API_PORT}"
export SCENARIO_AUTHENTICATOR_UI_URL="http://localhost:${AUTH_UI_PORT}"

Add these environment variables to your scenario’s runtime (for example via .env files or Makefile exports) so every service can reach the authenticator consistently.

2. Secure Your API Middleware

Express middleware example:

const authApiUrl = process.env.SCENARIO_AUTHENTICATOR_API_URL ?? 'http://localhost:15837';

export async function validateAuth(req, res, next) {
  const token = req.headers.authorization?.replace(/^Bearer\s+/i, '');
  if (!token) {
    return res.status(401).json({ error: 'Missing bearer token' });
  }

  try {
    const response = await fetch(`${authApiUrl}/api/v1/auth/validate`, {
      headers: { Authorization: `Bearer ${token}` }
    });

    const payload = await response.json();
    if (!payload.valid) {
      return res.status(401).json({ error: 'Token rejected by authenticator' });
    }

    req.user = {
      id: payload.user_id,
      email: payload.email,
      roles: payload.roles || []
    };
    return next();
  } catch (error) {
    console.error('[auth] validation failed', error);
    return res.status(503).json({ error: 'Authenticator unavailable' });
  }
}

Go validation helper:

func ValidateRequest(ctx context.Context, token string) error {
    authAPI := os.Getenv("SCENARIO_AUTHENTICATOR_API_URL")
    if authAPI == "" {
        authAPI = "http://localhost:15837"
    }

    req, err := http.NewRequestWithContext(ctx, http.MethodGet, authAPI+"/api/v1/auth/validate", nil)
    if err != nil {
        return err
    }
    req.Header.Set("Authorization", "Bearer "+token)

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        return err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return fmt.Errorf("token rejected: %s", resp.Status)
    }

    var payload struct {
        Valid  bool     `json:"valid"`
        UserID string   `json:"user_id"`
        Roles  []string `json:"roles"`
    }
    if err := json.NewDecoder(resp.Body).Decode(&payload); err != nil {
        return err
    }
    if !payload.Valid {
        return errors.New("token invalid")
    }

    // Attach payload metadata to the request context here.
    return nil
}

3. Protect Your UI Routes

Fetch the authenticator configuration (or reuse the exported environment variables) and redirect unauthenticated users to the hosted login screen.

async function requireAuth() {
  const token = localStorage.getItem('authToken');
  const config = window.__authConfig || (await fetch('/config').then((res) => res.json()).catch(() => ({ })));

  const authApiUrl = window.SCENARIO_AUTHENTICATOR_API_URL || config.apiUrl || 'http://localhost:15837';
  const authUiUrl = window.SCENARIO_AUTHENTICATOR_UI_URL || config.uiUrl || 'http://localhost:37578';

  if (!token || !authApiUrl) {
    window.location.href = `${authUiUrl}/login?redirect=${encodeURIComponent(window.location.href)}`;
    return false;
  }

  const response = await fetch(`${authApiUrl}/api/v1/auth/validate`, {
    headers: { Authorization: `Bearer ${token}` }
  });
  const validation = await response.json();

  if (!validation.valid) {
    localStorage.removeItem('authToken');
    window.location.href = `${authUiUrl}/login?redirect=${encodeURIComponent(window.location.href)}`;
    return false;
  }

  window.currentUser = validation;
  return true;
}

document.addEventListener('DOMContentLoaded', requireAuth);
